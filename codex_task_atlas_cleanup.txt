  <script>                                                                                                                                        
  const stepSel = document.getElementById('step');                                                                                                
  const layerSel = document.getElementById('layer');                                                                                              
  const img = document.getElementById('img');                                                                                                     
  const zoom = document.getElementById('zoom');                                                                                                   
  const legendEl = document.getElementById('legend');                                                                                             
  const status = document.getElementById('status');                                                                                               
                                                                                                                                                  
  const preferredLayerOrder = [                                                                                                                   
    "bands.png",                                                                                                                                  
    "temperature.png",                                                                                                                            
    "humidity.png",                                                                                                                               
    "continentalness.png",                                                                                                                        
    "erosion.png",                                                                                                                                
    "weirdness.png",                                                                                                                              
    "mask_swamp.png",                                                                                                                             
    "mask_mangrove.png",                                                                                                                          
    "biomes.png"                                                                                                                                  
  ];                                                                                                                                              
                                                                                                                                                  
  let index = null;                                                                                                                               
                                                                                                                                                  
  function applyZoom() {                                                                                                                          
    const z = Math.max(1, Math.min(8, parseInt(zoom.value || "1", 10)));                                                                          
    img.style.maxWidth = `min(85vw, ${1100*z}px)`;                                                                                                
  }                                                                                                                                               
                                                                                                                                                  
  function setImage(step, file) {                                                                                                                 
    img.src = `${step}/${file}?t=${Date.now()}`;                                                                                                  
    status.textContent = `Showing ${step}/${file}`;                                                                                               
  }                                                                                                                                               
                                                                                                                                                  
  async function loadLegend(step) {                                                                                                               
    // legend is per-step in your outputs                                                                                                         
    try {                                                                                                                                         
      const r = await fetch(`${step}/legend.json?t=${Date.now()}`);                                                                               
      if (!r.ok) throw new Error("no legend.json");                                                                                               
      const j = await r.json();
      legendEl.textContent = JSON.stringify(j, null, 2);                                                                                          
    } catch {                                                                                                                                     
      legendEl.textContent = "No legend.json in this step folder.";                                                                               
    }                                                                                                                                             
  }                                                                                                                                               
                                                                                                                                                  
  function populateLayersForStep(stepName) {                                                                                                      
    const step = index.find(x => x.step === stepName);                                                                                            
    const files = new Set(step.files);                                                                                                            
                                                                                                                                                  
    const found = [];                                                                                                                             
    for (const f of preferredLayerOrder) if (files.has(f)) found.push(f);
    // fallback: include any other pngs                                                                                                           
    for (const f of step.files) if (f.endsWith(".png") && !found.includes(f)) found.push(f);                                                      
                                                                                                                                                  
    layerSel.innerHTML = "";                                                                                                                      
    for (const f of found) {                                                                                                                      
      const opt = document.createElement("option");                                                                                               
      opt.value = f;                                                                                                                              
      opt.textContent = f.replace(".png","");                                                                                                     
      layerSel.appendChild(opt);                                                                                                                  
    }                                                                                                                                             
                                                                                                                                                  
    const first = found[0] || "biomes.png";                                                                                                       
    setImage(stepName, first);                                                                                                                    
    loadLegend(stepName);                                                                                                                         
  }                                                                                                                                               
                                                                                                                                                  
  async function init() {                                                                                                                         
    const r = await fetch(`atlas_index.json?t=${Date.now()}`);                                                                                    
    index = await r.json();                                                                                                                       
                                                                                                                                                  
    // Prefer step16 if it exists; else step8; else first step                                                                                    
    const steps = index.map(x => x.step);                                                                                                         
    const preferredStep = steps.includes("step16") ? "step16" : (steps.includes("step8") ? "step8" : steps[0]);                                   
                                                                                                                                                  
    stepSel.innerHTML = "";                                                                                                                       
    for (const s of steps) {                                                                                                                      
      const opt = document.createElement("option");                                                                                               
      opt.value = s;                                                                                                                              
      opt.textContent = s;                                                                                                                        
      stepSel.appendChild(opt);                                                                                                                   
    }                                                                                                                                             
    stepSel.value = preferredStep;                                                                                                                
                                                                                                                                                  
    populateLayersForStep(preferredStep);                                                                                                         
                                                                                                                                                  
    stepSel.addEventListener("change", () => populateLayersForStep(stepSel.value));
    layerSel.addEventListener("change", () => setImage(stepSel.value, layerSel.value));                                                           
    zoom.addEventListener("change", applyZoom);                                                                                                   
    document.getElementById("reload").addEventListener("click", () => loadLegend(stepSel.value));                                                 
                                                                                                                                                  
    applyZoom();                                                                                                                                  
  }                                                                                                                                               
                                                                                                                                                  
  init().catch(e => {                                                                                                                             
    status.textContent = "Failed to load atlas_index.json (use the local server step).";                                                          
  });                                                                                                                                             
  </script>
